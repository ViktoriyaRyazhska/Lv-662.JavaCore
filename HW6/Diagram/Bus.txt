package lecture7;

public class Bus extends GroundVehicle {
	String route;

	public Bus(int passengers, String route) {
		super(passengers);
		this.route = route;
	}

	@Override
	protected void drive() {
		super.drive();
	}

	public String getRoute() {
		return route;
	}

	public void setRoute(String route) {
		this.route = route;
	}

}
