package lecture7;

public class Liner extends WaterVehicle {
	private int floors;

	public Liner(int passengers, int floors) {
		super(passengers);
		this.floors = floors;

	}

	@Override
	protected void isSailing() {

		super.isSailing();
	}

	public int getFloors() {
		return floors;
	}

	public void setFloors(int floors) {
		this.floors = floors;
	}

}
