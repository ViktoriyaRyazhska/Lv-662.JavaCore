package lecture7;

public class Helicopter extends FlyingVehicle {

	private int weight;
	private int maxHeight;
	
	public Helicopter(int passengers, int weight, int maxHeight) {
		super(passengers);
		this.weight = weight;
		this.maxHeight = maxHeight;
	}

	@Override
	protected void fly() {
		super.fly();
	}

	@Override
	protected void land() {
		super.land();
	}


}
